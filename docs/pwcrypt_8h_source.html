209 static inline void dev_urandom(uint8_t* output_buffer, const size_t output_buffer_size)
210 {
211  if (output_buffer != NULL && output_buffer_size > 0)
212  {
213 #ifdef _WIN32
214  BCryptGenRandom(NULL, output_buffer, (ULONG)output_buffer_size, BCRYPT_USE_SYSTEM_PREFERRED_RNG);
215 #else
216  FILE* rnd = fopen("/dev/urandom", "r");
217  if (rnd != NULL)
218  {
219  const size_t n = fread(output_buffer, sizeof(unsigned char), output_buffer_size, rnd);
220  if (n != output_buffer_size)
221  {
222  pwcrypt_fprintf(stderr, "pwcrypt: Warning! Only %llu bytes out of %llu have been read from /dev/urandom\n", n, output_buffer_size);
223  }
224  fclose(rnd);
225  }
226 #endif
227  }
228 }

235 static inline size_t pwcrypt_get_filesize(const char* filepath)
236 {
237  size_t filesize = 0;
238 
239 #if _WIN32
240  HANDLE f = CreateFile(filepath, FILE_ATTRIBUTE_READONLY, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
241  if (f == INVALID_HANDLE_VALUE)
242  {
243  pwcrypt_fprintf(stderr, "pwcrypt: Failure to open file for reading its size: %s", filepath);
244  goto exit;
245  }
246 
247  LARGE_INTEGER i;
248  if (!GetFileSizeEx(f, &i))
249  {
250  pwcrypt_fprintf(stderr, "pwcrypt: Failure to read file size: %s", filepath);
251  goto exit;
252  }
253 
254  filesize = (size_t)i.QuadPart;
255 
256 exit:
257  CloseHandle(f);
258  return filesize;
259 #else
260  struct stat64 stbuf;
261  int fd = open(filepath, 0x00);
262  if (fd == -1)
263  {
264  pwcrypt_fprintf(stderr, "pwcrypt: Failure to open file for reading its size: %s", filepath);
265  goto exit;
266  }
267 
268  if ((fstat64(fd, &stbuf) != 0) || (!S_ISREG(stbuf.st_mode)))
269  {
270  pwcrypt_fprintf(stderr, "pwcrypt: Failure to retrieve filesize: %s", filepath);
271  goto exit;
272  }
273 
274  filesize = (size_t)stbuf.st_size;
275 
276 exit:
277  close(fd);
278  mbedtls_platform_zeroize(&stbuf, sizeof(struct stat64));
279  return filesize;
280 #endif
281 }
