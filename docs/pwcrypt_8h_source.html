170 static inline void dev_urandom(uint8_t* output_buffer, const size_t output_buffer_size)
171 {
172  if (output_buffer != NULL && output_buffer_size > 0)
173  {
174 #ifdef _WIN32
175  BCryptGenRandom(NULL, output_buffer, (ULONG)output_buffer_size, BCRYPT_USE_SYSTEM_PREFERRED_RNG);
176 #else
177  FILE* rnd = fopen("/dev/urandom", "r");
178  if (rnd != NULL)
179  {
180  const size_t n = fread(output_buffer, sizeof(unsigned char), output_buffer_size, rnd);
181  if (n != output_buffer_size)
182  {
183  pwcrypt_fprintf(stderr, "pwcrypt: Warning! Only %llu bytes out of %llu have been read from /dev/urandom\n", n, output_buffer_size);
184  }
185  fclose(rnd);
186  }
187 #endif
188  }
189 }
