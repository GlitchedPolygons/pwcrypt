104 static inline void dev_urandom(uint8_t* output_buffer, const size_t output_buffer_size)
105 {
106  if (output_buffer != NULL && output_buffer_size > 0)
107  {
108 #ifdef _WIN32
109  BCryptGenRandom(NULL, output_buffer, output_buffer_size, BCRYPT_USE_SYSTEM_PREFERRED_RNG);
110 #else
111  FILE* rnd = fopen("/dev/urandom", "r");
112  if (rnd != NULL)
113  {
114  fread(output_buffer, sizeof(unsigned char), output_buffer_size, rnd);
115  fclose(rnd);
116  }
117 #endif
118  }
119 }
