157 static inline void dev_urandom(uint8_t* output_buffer, const size_t output_buffer_size)
158 {
159  if (output_buffer != NULL && output_buffer_size > 0)
160  {
161 #ifdef _WIN32
162  BCryptGenRandom(NULL, output_buffer, (ULONG)output_buffer_size, BCRYPT_USE_SYSTEM_PREFERRED_RNG);
163 #else
164  FILE* rnd = fopen("/dev/urandom", "r");
165  if (rnd != NULL)
166  {
167  const size_t n = fread(output_buffer, sizeof(unsigned char), output_buffer_size, rnd);
168  if (n != output_buffer_size)
169  {
170  pwcrypt_fprintf(stderr, "pwcrypt: Warning! Only %llu bytes out of %llu have been read from /dev/urandom\n", n, output_buffer_size);
171  }
172  fclose(rnd);
173  }
174 #endif
175  }
176 }
